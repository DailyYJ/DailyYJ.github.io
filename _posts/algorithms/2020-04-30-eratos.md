public boolean[] eratos(int num) {
    boolean[] result = new boolean[num + 1];
    for (int i = 2; i < Math.sqrt(num); i++) {
        if (!result[i]) {
            for (int j = i * 2; j <= num; j += i) {
                result[j] = true;
            }
        }
    }
    return result;
}
